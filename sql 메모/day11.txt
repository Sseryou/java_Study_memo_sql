CREATE TABLE 테이블명(
	컬럼명 자료형 [제약조건],
	...

);

자료형
문자 = 
	CHAR
	VARCHAR2
	NVARCHAR2
	CLOB

숫자 =
	NUMBER(수치, 실수) 수치 -> 정수 + 소수점 자리

날짜 = 
	DATE

제약조건
	NOT NULL
	UNIQUE - 중복값X, NULL은 허용
	PRIMARY KEY - UNIQUE + NOT NULL
	FOREIGN KEY - REPERENCE 테이블명(컬럼명)
	CHECK(조건식)
	DEFAULT 기본값

ALTER TABLE 테이블명
		ADD 컬럼명 자료형 [제약조건] : 컬럼 추가
		DROP COLUMN 컬럼명;
		MODIFY
		RENAME COLUMN 컬럼명 TO 새로운 컬럼명;

CREATE SEQUENCE 시퀀스명;
	IDENTIFIED BY 수치,
	START WITH
	MINVALUE
	MAXVALUE
	CYCLE | NOCYCLE
	CACHE n | NO CACHE;

권한
시스템 권한
	ALL PRIVILEGES -> 모든 권한
	GRANT 시스템 권한, .... TO 사용자 [WITH ADMIN OPTION];

	REVOKE 시스템 권한 ... TO 사용자;


객체 권한
	ALL PRIVILEGES -> 모든 권한
	GRANT 객체 권한(SELECT, UPDATE, INSERT...) ON 사용자명.테이블명 TO 사용자 [WITH GRANT OPTION]

	REVOKE 객체 권한 ON ... FROM 사용자;
